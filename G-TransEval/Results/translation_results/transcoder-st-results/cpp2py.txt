def DoAlgebra ( ops , operands ) :
    nums = list ( operands )
    current_ops = list ( ops )
    i = len ( current_ops ) - 1
    while i >= 0 :
        if current_ops [ i ] == '**' :
            nums [ i ] = math.pow ( nums [ i ] , nums [ i + 1 ] )
            nums.pop ( i + 1 )
            current_ops.pop ( i )
        i -= 1
    i = 0
    while i < len ( current_ops ) :
        if current_ops [ i ] == '*' :
            nums [ i ] *= nums [ i + 1 ]
            nums.pop ( i + 1 )
            current_ops.pop ( i )
        elif current_ops [ i ] == '/' :
            nums [ i ] /= nums [ i + 1 ]
            nums.pop ( i + 1 )
            current_ops.pop ( i )
        else :
            i += 1
    i = 0
    while i < len ( current_ops ) :
        if current_ops [ i ] == '+' :
            nums [ i ] += nums [ i + 1 ]
            nums.pop ( i + 1 )
            current_ops.pop ( i )
        elif current_ops [ i ] == '-' :
            nums [ i ] -= nums [ i + 1 ]
            nums.pop ( i + 1 )
            current_ops.pop ( i )
        else :
            i += 1
    return nums [ 0 ]

                if t [ k ] != '0' :
                    pos = k
                    break
            for k in range ( pos , 0 , - 1 ) :
                t [ k ] = t [ k - 1 ]
                cur += 1
            nn = int ( t )
            if nn % 25 == 0 :
                ans = min ( ans , cur )
    if ans == sys.maxsize :
        return - 1
    return ans

def KaprekarRec ( n , prev ) :
    if n == 0 :
        return 0
    prev = n
    digits = [ n % 10 for i in range ( 4 ) ]
    digits.sort ( )
    asc = 0
    for i in range ( 4 ) :
        asc = asc * 10 + digits [ i ]
    digits.sort ( )
    desc = 0
    for i in range ( 3 , - 1 , - 1 ) :
        desc = desc * 10 + digits [ i ]
    diff = abs ( asc - desc )
    if diff == prev :
        return diff
    return KaprekarRec ( diff , prev )

def FractionToDecimal ( numr , denr ) :
    res = ''
    mp = { }
    rem = numr % denr
    while rem != 0 and mp.get ( rem ) == 0 :
        mp [ rem ] = len ( res )
        rem = rem * 10
        res_part = rem // denr
        res += res_part
        rem = rem % denr
    if rem == 0 :
        return ''
    if mp.get ( rem ) != 0 :
        return res [ mp [ rem ] : ]
    return ''

def is_ancy ( num ) :
    fp = { '0' : '0' , '1' : '1' , '6' : '9' , '8' : '8' , '9' : '6' }
    n = len ( num )
    l , r = 0 , n - 1
    while l <= r :
        if fp.get ( num [ l ] ) is None or fp [ num [ l ] ] != num [ r ] :
            return False
        l += 1
        r -= 1
    return True
